  def cell(row_addr, col_addr)
    check_frame_constraint(row_addr, col_addr)

    a_cell = find_cell(row_addr, col_addr)
    unless a_cell
      a_cell = Cell.new(row_addr, col_addr)
      @cells.push a_cell
    end
    # merge_rule‚ð“K—p
    a_cell.merge_rule(@merge_rule)
    a_cell
  end
