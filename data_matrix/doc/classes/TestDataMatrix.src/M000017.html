  def test01_cell01
    cell1 = @dm.cell("A","1")
    
    # cell には アドレスが設定されている。
    assert_equal('A',cell1.row_addr)
    assert_equal('1',cell1.col_addr)

    # 同じアドレスには同じCellオブジェクトが返却される。
    cell2 = @dm.cell("A","1")

    assert_equal(cell1.object_id, cell2.object_id)
  end
